## ANSWER:

``` r
# goal of function: read each .pdb file, then trim the .pdb file to pick out rows where chain is A and elety is CA

#input: name of pdb file
func <- function(x="4AKE") {                 
  s <- read.pdb(x)
  s.chainA <- trim.pdb(s, chain="A", elety="CA")
  s.b <- s.chainA$atom$b
  plotb3(s.b, sse=s.chainA, type ="l", ylab="Bfactor")
}
```
